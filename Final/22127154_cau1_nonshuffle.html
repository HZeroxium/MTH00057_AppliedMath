def train_test_split(
    X: Matrix, y: Matrix, test_size: float = 0.2, shuffle: bool = True
) -> Tuple[Matrix, Matrix, Matrix, Matrix]:
    """Split the data into training and testing sets.

    Args:
        X (Matrix): Feature matrix.
        y (Matrix): Target matrix.
        test_size (float, optional): Proportion of the dataset to include in the test split. Defaults to 0.2.
        shuffle (bool, optional): Whether to shuffle the data before splitting. Defaults to True.

    Returns:
        Tuple[Matrix, Matrix, Matrix, Matrix]: The training and testing sets for both features and target.
    """
    indices = list(range(X.rows))

    if shuffle:
        random.shuffle(indices)

    test_indices = indices[: math.floor(test_size * len(indices))]
    train_indices = indices[math.floor(test_size * len(indices)) :]

    X_train = Matrix([X[i] for i in train_indices])
    y_train = Matrix([y[i] for i in train_indices])
    X_test = Matrix([X[i] for i in test_indices])
    y_test = Matrix([y[i] for i in test_indices])

    return X_train, X_test, y_train, y_test

def evaluate_model(
    model: LinearRegression,
    X_train: Matrix,
    y_train: Matrix,
    X_test: Matrix,
    y_test: Matrix,
) -> float:
    """Train the model and evaluate it using Mean Squared Error (MSE).

    Args:
        model (LinearRegression): The regression model to train.
        X_train (Matrix): The training feature matrix.
        y_train (Matrix): The training target matrix.
        X_test (Matrix): The testing feature matrix.
        y_test (Matrix): The testing target matrix.

    Returns:
        float: The Mean Squared Error (MSE) of the model on the test set.
    """
    model.fit(X_train, y_train)
    y_pred = model.predict(X_test)
    mse = model.mean_squared_error(y_test, y_pred)
    return mse

4.5. Xây dựng mô hình hồi quy tuyến tính¶

Trong phần này, em sẽ xây dựng mô hình hồi quy tuyến tính với các yêu cầu đã nêu. Em đã xây dựng được một số mô hình như sau:

  • LinearRegression: Mô hình hồi quy tuyến tính bậc nhất. $$y = w_0 + w_1x_1 + w_2x_2 + ... + w_nx_n$$
  • PolynomialRegression: Mô hình hồi quy tuyến tính bậc cao. $$y = w_0 + w_1x_1 + w_2x_2^2 + ... + w_nx_n^n$$

Ví dụ: Với bậc 2, mô hình sẽ trở thành: $$y = w_0 + w_{11}x1 + w_{12}x1^2 + w_{21}x2 + w_{22}x2^2 + ... + w_{n1}x_n + w_{n2}x_n^2$$

Với bậc 3, mô hình sẽ trở thành: $$y = w_0 + w_{11}x^1 + w_{12}x_1^2 + w_{13}x_1^3 + w_{21}x_2 + w_{22}x_2^2 + w_{23}x_2^3 + ... + w_{n1}x_n + w_{n2}x_n^2 + w_{n3}x_n^3$$

  • InteractionTermsRegression: Mô hình hồi quy tuyến tính với các biến tương tác. $$y = w_0 + w_1x_1 + w_2x_2 + ... + w_nx_n + w_{12}x_1x_2 + w_{13}x_1x_3 + ... + w_{1n}x_1x_n + w_{23}x_2x_3 + ... + w_{2n}x_2x_n + ... + w_{(n-1)n}x_{(n-1)}x_n$$

  • RidgeRegression: Mô hình hồi quy tuyến tính với regularization L2. $$y = w_0 + w_1x_1 + w_2x_2 + ... + w_nx_n + \lambda \sum_{i=1}^{n}w_i^2$$

  • LassoRegression: Mô hình hồi quy tuyến tính với regularization L1. $$y = w_0 + w_1x_1 + w_2x_2 + ... + w_nx_n + \lambda \sum_{i=1}^{n}|w_i|$$

  • ElasticNetRegression: Mô hình hồi quy tuyến tính với regularization L1 và L2. $$y = w_0 + w_1x_1 + w_2x_2 + ... + w_nx_n + \lambda_1 \sum_{i=1}^{n}w_i^2 + \lambda_2 \sum_{i=1}^{n}|w_i|$$

  • BayesianRegression: Mô hình hồi quy tuyến tính với phân phối Gaussian. $$y = w_0 + w_1x_1 + w_2x_2 + ... + w_nx_n + \epsilon, \epsilon \sim N(0, \sigma^2)$$

Tuy nhiên, trong quá trình chạy thử nghiệm, em thấy các mô hình PolynomialRegression, LassoRegression, ElasticNetRegression cho kết quả với sai số rất lớn, nên em sẽ không sử dụng mô hình này trong bài toán này.

def run_models_with_different_splits(
    X: DataFrame,
    y: Series,
    split_ratios: List[float] = [0.1, 0.2, 0.4, 0.6],
    num_runs: int = 10,
    models: Dict[str, LinearRegression] = {
        "Linear Regression": LinearRegression(),
    },
    plot_result: bool = True,
    shuffle: bool = True,
    print_sub_table: bool = True,
) -> None:
    """Run models with different train/test splits and visualize the results.

    Args:
        X (DataFrame): The feature matrix.
        y (Series): The target vector.
        split_ratios (List[float], optional): List of split ratios to use. Defaults to [0.1, 0.2, 0.4, 0.6].
        num_runs (int, optional): Number of runs for each split. Defaults to 10.
        models (Dict[str, LinearRegression], optional): Dictionary of models to evaluate. Defaults to {"Linear Regression": LinearRegression()}.
        plot_result (bool, optional): Whether to plot the results. Defaults to True.
        shuffle (bool, optional): Whether to shuffle the data before splitting. Defaults to True.
        print_sub_table (bool, optional): Whether to print sub-tables for each model and split. Defaults to True.
    """
    if not shuffle:
        num_runs = 1
    X_list = X.to_numpy().tolist()
    y_list = y.to_numpy().tolist()
    X_matrix = Matrix(X_list)
    y_matrix = Matrix([y_list]).transpose()
    results = {model_name: [] for model_name in models}

    for split_ratio in split_ratios:
        table_data = []
        headers = ["Model", "Split Ratio", "MSE"]

        for model_name, model in models.items():
            # print(
            #     f"Running {model_name} with {int((1 - split_ratio) * 100)}/{int(split_ratio * 100)} train/test split..."
            # )
            mse_list = []
            for _ in range(num_runs):
                X_train, X_test, y_train, y_test = train_test_split(
                    X_matrix, y_matrix, test_size=split_ratio, shuffle=shuffle
                )
                mse = evaluate_model(model, X_train, y_train, X_test, y_test)
                mse_list.append(mse)

            avg_mse = sum(mse_list) / num_runs
            results[model_name].append(avg_mse)
            table_data.append(
                [
                    model_name,
                    f"{int((1 - split_ratio) * 100)}/{int(split_ratio * 100)}",
                    f"{avg_mse:.4f}",
                ]
            )
        if print_sub_table:
            print(tabulate(table_data, headers=headers, tablefmt="pretty"))

    formatted_results = format_results(results)

    # Plot the results for better visualization
    if plot_result:
        plot_results(
            results,
            "Mean Squared Error (MSE) across Different Models and Train/Test Splits",
        )

    # Tìm mô hình tốt nhất dựa trên MSE trung bình
    avg_results = {
        model_name: sum(mse_list) / len(mse_list)
        for model_name, mse_list in results.items()
    }
    best_model_name = min(avg_results, key=avg_results.get)

    print(
        f"=====> a) Best model: {best_model_name} with average MSE: {avg_results[best_model_name]:.4f}"
    )

def run_models_with_single_feature(
    X: DataFrame,
    y: Series,
    features: List[str],
    split_ratios: List[float] = [0.1, 0.2, 0.4, 0.6],
    num_runs: int = 10,
    models: Dict[str, LinearRegression] = {
        "Linear Regression": LinearRegression(),
    },
    plot_result: bool = True,
    shuffle: bool = True,
    print_sub_table: bool = True,
) -> None:
    """Run models with different train/test splits for each single feature.

    Args:
        X (DataFrame): The feature matrix.
        y (Series): The target vector.
        features (List[str]): List of feature names to evaluate individually.
        split_ratios (List[float], optional): List of split ratios to use. Defaults to [0.1, 0.2, 0.4, 0.6].
        num_runs (int, optional): Number of runs for each split. Defaults to 10.
        models (Dict[str, LinearRegression], optional): Dictionary of models to evaluate. Defaults to {"Linear Regression": LinearRegression()}.
        plot_result (bool, optional): Whether to plot the results. Defaults to True.
        shuffle (bool, optional): Whether to shuffle the data before splitting. Defaults to True.
        print_sub_table (bool, optional): Whether to print sub-tables for each model and split. Defaults to True.
    """
    if not shuffle:
        num_runs = 1
    models.pop(
        "Interaction Terms Regression", None
    )  # Remove model that requires multiple features
    y_list = y.to_numpy().tolist()
    y_matrix = Matrix([y_list]).transpose()

    final_results = {}

    for feature_idx, feature in enumerate(features):
        table_data = []
        headers = ["Model", "Feature", "Split Ratio", "MSE"]

        X_single_feature = X[feature]  # Only use one feature column
        X_list = X_single_feature.to_numpy().tolist()
        X_matrix = Matrix([X_list]).transpose()
        results = {model_name: [] for model_name in models}

        for split_ratio in split_ratios:
            for model_name, model in models.items():
                mse_list = []
                for _ in range(num_runs):
                    X_train, X_test, y_train, y_test = train_test_split(
                        X_matrix, y_matrix, test_size=split_ratio, shuffle=shuffle
                    )
                    mse = evaluate_model(model, X_train, y_train, X_test, y_test)
                    mse_list.append(mse)

                avg_mse = sum(mse_list) / num_runs
                results[model_name].append(avg_mse)
                table_data.append(
                    [
                        model_name,
                        feature,
                        f"{int((1 - split_ratio) * 100)}/{int(split_ratio * 100)}",
                        f"{avg_mse:.4f}",
                    ]
                )
        if print_sub_table:
            print(tabulate(table_data, headers=headers, tablefmt="pretty"))

        final_results[feature] = results

    # Plot results for each feature
    if plot_result:
        for feature, result in final_results.items():
            formatted_results = format_results(result)
            plot_results(result, f"Mean Squared Error (MSE) for {feature}")

    # Find the best model for each feature
    best_results = {}
    for feature, result in final_results.items():
        avg_results = {
            model_name: sum(mse_list) / len(mse_list)
            for model_name, mse_list in result.items()
        }
        best_model_name = min(avg_results, key=avg_results.get)
        best_results[feature] = (best_model_name, avg_results[best_model_name])

    summary_table = []
    for feature, (model_name, mse) in best_results.items():
        summary_table.append([feature, model_name, f"{mse:.4f}"])

    print(
        tabulate(
            summary_table, headers=["Feature", "Best Model", "MSE"], tablefmt="pretty"
        )
    )

    # Find the best overall feature and model
    best_feature = min(best_results, key=lambda x: best_results[x][1])
    best_model, best_mse = best_results[best_feature]
    print(
        f"\n=======> b) Best feature: {best_feature} | Best Model: {best_model} | MSE: {best_mse:.4f}"
    )

def run_models_with_feature_combinations(
    X: DataFrame,
    y: Series,
    features: List[str],
    split_ratios: List[float] = [0.1, 0.2, 0.4, 0.6],
    num_runs: int = 10,
    models: Dict[str, LinearRegression] = {
        "Linear Regression": LinearRegression(),
    },
    plot_result: bool = True,
    shuffle: bool = True,
    print_sub_table: bool = True,
) -> None:
    """Run models with different train/test splits for all feature combinations.

    Args:
        X (DataFrame): The feature matrix.
        y (Series): The target vector.
        features (List[str]): List of feature names to combine and evaluate.
        split_ratios (List[float], optional): List of split ratios to use. Defaults to [0.1, 0.2, 0.4, 0.6].
        num_runs (int, optional): Number of runs for each split. Defaults to 10.
        models (Dict[str, LinearRegression], optional): Dictionary of models to evaluate. Defaults to {"Linear Regression": LinearRegression()}.
        plot_result (bool, optional): Whether to plot the results. Defaults to True.
        shuffle (bool, optional): Whether to shuffle the data before splitting. Defaults to True.
        print_sub_table (bool, optional): Whether to print sub-tables for each model and split. Defaults to True.
    """
    if not shuffle:
        num_runs = 1
    y_list = y.to_numpy().tolist()
    y_matrix = Matrix([y_list]).transpose()

    final_results = {}

    for r in range(
        1, len(features) + 1
    ):  # Generate all combinations from 1 to 7 features
        for combination in combinations(features, r):
            table_data = []
            headers = ["Model", "Feature Combination", "Split Ratio", "MSE"]

            if print_sub_table:
                print(f"\nEvaluating combination: {combination}")
            X_combination = X[
                list(combination)
            ]  # Only use the selected combination of features
            X_list = X_combination.to_numpy().tolist()
            X_matrix = Matrix(X_list)
            results = {model_name: [] for model_name in models}

            for split_ratio in split_ratios:
                for model_name, model in models.items():
                    mse_list = []
                    for _ in range(num_runs):
                        X_train, X_test, y_train, y_test = train_test_split(
                            X_matrix, y_matrix, test_size=split_ratio, shuffle=shuffle
                        )
                        mse = evaluate_model(model, X_train, y_train, X_test, y_test)
                        mse_list.append(mse)

                    avg_mse = sum(mse_list) / num_runs
                    results[model_name].append(avg_mse)
                    table_data.append(
                        [
                            model_name,
                            str(combination),
                            f"{int((1 - split_ratio) * 100)}/{int(split_ratio * 100)}",
                            f"{avg_mse:.4f}",
                        ]
                    )

            if print_sub_table:
                print(tabulate(table_data, headers=headers, tablefmt="pretty"))

            final_results[combination] = results

    # Plot results for each combination
    if plot_result:
        for combination, result in final_results.items():
            formatted_results = format_results(result)
            plot_results(result, f"Mean Squared Error (MSE) for {combination}")

    # Find the best model for each combination
    best_results = {}
    for combination, result in final_results.items():
        avg_results = {
            model_name: sum(mse_list) / len(mse_list)
            for model_name, mse_list in result.items()
        }
        best_model_name = min(avg_results, key=avg_results.get)
        best_results[combination] = (best_model_name, avg_results[best_model_name])

    # Print the best result for each combination
    summary_table = []
    for combination, (model_name, mse) in best_results.items():
        summary_table.append([str(combination), model_name, f"{mse:.4f}"])

    print(
        tabulate(
            summary_table,
            headers=["Combination", "Best Model", "MSE"],
            tablefmt="pretty",
        )
    )

    # Find the overall best combination
    best_combination = min(best_results, key=lambda x: best_results[x][1])
    best_model, best_mse = best_results[best_combination]
    print(
        f"\n========> c) Best combination: {best_combination} | Best Model: {best_model} | MSE: {best_mse:.4f}\n\n\n"
    )

4.5.4. Hàm chính¶

Cuối cùng, em sẽ viết hàm chính để thực hiện các yêu cầu đã nêu ở trên. Hàm này sẽ gọi lần lượt các hàm đã viết ở các phần trước để thực hiện các yêu cầu đã nêu.

Mỗi hàm đều có thể tùy chỉnh thông số để thực hiện các yêu cầu khác nhau tùy theo yêu cầu của thầy cô. Thầy cô có thể chỉnh thông số ở các biến global ở phần 3 hoặc chỉnh trực tiếp trong hàm chính. Mặc định, em đã thiết lập các thông số mặc định như sau:

  • split_ratios: [0.4, 0.6, 0.8, 0.9]: 4 tỉ lệ chia dữ liệu khác nhau. Thầy cô có thể chỉnh tỉ lệ chia dữ liệu khác nhau.
  • shuffle: True: Thầy cô có thể chỉnh False để không xáo trộn dữ liệu trước khi chia dữ liệu.
  • num_runs: 10: Thầy cô có thể chỉnh số lần chạy mô hình để đánh giá độ ổn định của mô hình.
  • plot_result: True: Thầy cô có thể chỉnh False để không trực quan hóa kết quả.
  • print_sub_table: False: Thầy cô có thể chỉnh True để in ra bảng kết quả chi tiết của từng model với mỗi tỉ lệ chia dữ liệu.

Tuy nhiên, riêng hàm run_models_with_feature_combinations, em đặt mặc định là không in ra bảng kết quả chi tiết của từng model với mỗi tỉ lệ chia dữ liệu vì số lượng tổ hợp đặc trưng quá lớn. Thầy cô có thể chỉnh thông số print_sub_table để in ra bảng kết quả chi tiết và plot_result để in ra biểu đồ so sánh của từng model với mỗi tỉ lệ chia dữ liệu.

5. Kết luận¶

Sau quá trình thực hiện bài toán, em đã xây dựng mô hình hồi quy tuyến tính với các yêu cầu đã nêu. Kết quả mỗi lần chạy có thể thay đổi tùy thuộc vào nhiều yếu tố: việc chia dữ liệu ngẫu nhiên, việc xáo trộn dữ liệu, việc chọn mô hình, việc chọn đặc trưng, số lần chạy của một mô hình. Điều này là tự nhiên và không thể tránh khỏi. Tuy nhiên, việc chạy nhiều lần giúp đánh giá độ ổn định của mô hình và đưa ra kết luận chính xác hơn.
